local function GetAnim(s)
    if AnimCache[s] then return AnimCache[s] end
    if A.Dribbles[s] then
        local a = Instance.new("Animation")
        a.AnimationId = A.Dribbles[s]
        local track = M:LoadAnimation(a)
        AnimCache[s] = track
        return track
    end
end

local function IsOpponent(p)
    return L.Team and p.Team and L.Team ~= p.Team
end

-- Script originally made by denzel.natsuki on discord

local slideWasActive = {}
local touchCooldown  = {}
local TOUCH_COOLDOWN = 0.6

local function DribbleAnim()
    if not (C.Values and C.Values.HasBall.Value) then return end
    B:FireServer()
    local style = L.PlayerStats.Style.Value
    local track = GetAnim(style)
    if track and not track.IsPlaying then
        track:Play()
        track:AdjustSpeed(1)
    end
    local ball = workspace:FindFirstChild("Football")
    if ball then
        ball.AssemblyLinearVelocity = Vector3.zero
        ball.CFrame = H.CFrame * CFrame.new(0, -2.5, 0)
    end
end

local function DribbleNoAnim()
    if not (C.Values and C.Values.HasBall.Value) then return end
    B:FireServer()
    local ball = workspace:FindFirstChild("Football")
    if ball then
        ball.AssemblyLinearVelocity = Vector3.zero
        ball.CFrame = H.CFrame * CFrame.new(0, -2.5, 0)
    end
end

local function FireDribble()
    if S.EnabledAnim   then DribbleAnim()   end
    if S.EnabledNoAnim then DribbleNoAnim() end
end

-- Returns current round-trip ping in seconds, capped to avoid bad values
local function GetPing()
    local ok, ping = pcall(function()
        return Players.LocalPlayer:GetNetworkPing()
    end)
    if ok and ping then
        return math.clamp(ping, 0.01, 0.4)
    end
    return 0.08 -- fallback 80ms if unavailable
end

local rangeSq      = S.range * S.range
local touchRangeSq = S.touchRange * S.touchRange

RunService.Heartbeat:Connect(function()
    if not (C and H) then return end
    if not (S.EnabledAnim or S.EnabledNoAnim) then return end
    if not (C.Values and C.Values.HasBall.Value) then return end

    local myPos = H.Position
    local now   = tick()
    local ping  = GetPing()

    -- Buffer: ping + one heartbeat (~16ms) + 30ms input latency
    local predTime = ping + 0.046

    for _, p in ipairs(Players:GetPlayers()) do
        if IsOpponent(p) and p.Character then
            local c = p.Character
            local v = c.Values and c.Values.Sliding
            local isSliding = v and v.Value == true
            local r = c:FindFirstChild("HumanoidRootPart")
            if r then
                local rPos = r.Position
                local rVel = r.AssemblyLinearVelocity

                -- Predict where the opponent WILL BE after ping+buffer seconds
                local predPos = rPos + rVel * predTime

                local diff        = rPos - myPos
                local distSq      = diff.X^2 + diff.Y^2 + diff.Z^2

                local predDiff    = predPos - myPos
                local predDistSq  = predDiff.X^2 + predDiff.Y^2 + predDiff.Z^2

                -- Use the SMALLER of current and predicted dist so we trigger
                -- if they are approaching fast (arriving within ping window)
                local effectiveDistSq = math.min(distSq, predDistSq)

                local inRange    = effectiveDistSq < rangeSq
                local isTouching = effectiveDistSq < touchRangeSq

                -- Slide trigger: rising edge, ping-compensated range
                if isSliding and not slideWasActive[p] and inRange then
                    slideWasActive[p] = true
                    FireDribble()

                -- Contact trigger: opponent is or will be literally on top of us
                elseif isTouching and (not touchCooldown[p] or now - touchCooldown[p] > TOUCH_COOLDOWN) then
                    touchCooldown[p] = now
                    FireDribble()
                end

                if not isSliding then slideWasActive[p] = false end
            end
        end
    end
end)

local function Notify(title, text, dur)
    StarterGui:SetCore("SendNotification", {
        Title    = title,
        Text     = text,
        Duration = dur or 3
    })
end

local animToggleRef   = nil
local noAnimToggleRef = nil

UserInputService.InputBegan:Connect(function(input, gp)
    if gp then return end
    if input.KeyCode == Enum.KeyCode.U then
        if S.EnabledAnim then
            S.EnabledAnim = false
            if animToggleRef then animToggleRef.SetState(false) end
            Notify("Auto Dribble (Anim)", "Auto Dribbling is Disabled")
        elseif S.EnabledNoAnim then
            S.EnabledNoAnim = false
            if noAnimToggleRef then noAnimToggleRef.SetState(false) end
            Notify("Auto Dribble (No Anim)", "Auto Dribbling is Disabled")
        end
    end
end)

local function createTab(name)
    local btn = Instance.new("TextButton")
    btn.Size             = UDim2.new(0, 0, 1, -4)
    btn.AutomaticSize    = Enum.AutomaticSize.X
    btn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    btn.TextColor3       = Color3.fromRGB(160, 160, 160)
    btn.TextSize         = 12
    btn.Font             = Enum.Font.GothamBold
    btn.Text             = " " .. name .. " "
    btn.BorderSizePixel  = 0
    btn.Parent           = tabBar

    local bc = Instance.new("UICorner")
    bc.CornerRadius = UDim.new(0, 4)
    bc.Parent       = btn

    local page = Instance.new("ScrollingFrame")
    page.Name                   = name
    page.Size                   = UDim2.new(1, 0, 1, 0)
    page.BackgroundTransparency = 1
    page.BorderSizePixel        = 0
    page.ScrollBarThickness     = 3
    page.ScrollBarImageColor3   = Color3.fromRGB(80, 80, 80)
    page.Visible                = false
    page.CanvasSize             = UDim2.new(0, 0, 0, 0)
    page.AutomaticCanvasSize    = Enum.AutomaticSize.Y
    page.Parent                 = contentArea

    local pageLayout = Instance.new("UIListLayout")
    pageLayout.FillDirection = Enum.FillDirection.Vertical
    pageLayout.Padding       = UDim.new(0, 6)
    pageLayout.Parent        = page

    local pagePadding = Instance.new("UIPadding")
    pagePadding.PaddingTop   = UDim.new(0, 8)
    pagePadding.PaddingLeft  = UDim.new(0, 10)
    pagePadding.PaddingRight = UDim.new(0, 10)
    pagePadding.Parent       = page

    tabs[name]     = btn
    tabPages[name] = page

-- Script originally made by denzel.natsuki on discord

    btn.MouseButton1Click:Connect(function()
        for n, b in pairs(tabs) do
            b.BackgroundColor3  = Color3.fromRGB(30, 30, 30)
            b.TextColor3        = Color3.fromRGB(160, 160, 160)
            tabPages[n].Visible = false
        end
        btn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
        btn.TextColor3       = Color3.fromRGB(255, 255, 255)
        page.Visible         = true
        activeTab            = name
    end)

    return page
end

local function selectTab(name)
    for n, b in pairs(tabs) do
        b.BackgroundColor3  = Color3.fromRGB(30, 30, 30)
        b.TextColor3        = Color3.fromRGB(160, 160, 160)
        tabPages[n].Visible = false
    end
    if tabs[name] then
        tabs[name].BackgroundColor3  = Color3.fromRGB(45, 45, 45)
        tabs[name].TextColor3        = Color3.fromRGB(255, 255, 255)
        tabPages[name].Visible       = true
        activeTab                    = name
    end
end

local openTime = tick()

RunService.RenderStepped:Connect(function(dt)
    local elapsed = tick() - openTime
    local mins    = math.floor(elapsed / 60)
    local secs    = math.floor(elapsed % 60)
    timerLabel.Text = string.format("Time Open: %02d:%02d", mins, secs)
    local fps = dt > 0 and math.floor(1 / dt) or 0
    fpsLabel.Text = string.format("FPS: %d", fps)
end)
